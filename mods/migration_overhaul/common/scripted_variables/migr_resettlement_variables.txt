# Timed forced resettlement tuning. See docs/ROADMAP.md (Population & Migration) and the
# mod README. Resettlement is instant + cheap in vanilla; this makes intra-empire (manual/
# forced) resettlement carry BOTH a time cost (a settling-in productivity penalty on the
# moved pops) and a resource SURCHARGE that scales by pops moved and by civics/traits/ethics.
#
# Everything here is applied event-side from on_pop_group_resettled (no polling, no vanilla
# override). Vanilla's own flat cost (100 energy + 10 unity, already scaled by the native
# pop_resettlement_cost_mult modifier) still applies up-front as the gate; this is additive.

# --- Resource surcharge (per pop, NEGATIVE = removed). Scaled by the disruption factor
#     below, which already folds in the pop count. ---
@migr_resettle_surcharge_energy	= -20
@migr_resettle_surcharge_unity	= -5

# --- Disruption factor (multiplies the surcharge). Tiered by how mobility-oriented the
#     resettling empire / species is. A factor of 1.0 = baseline settled empire. ---
# NB on corvee: vanilla civic_corvee_system does NOT remove resettlement cost — it gives
# pop_resettlement_cost_mult = -0.1 and waives only the UNITY portion (energy still paid).
# A gestalt civic gives -0.50. So we discount these but don't zero them.
@migr_resettle_factor_base		= 1.0	# default (non-gestalt, no resettlement civics)
@migr_resettle_factor_gestalt	= 0.4	# drones reassign with minimal social disruption (cf. gestalt civic -0.50)
@migr_resettle_factor_corvee	= 0.5	# civic_corvee_system / Adaptability finisher: built for it
# Species mobility traits scale further (multiplicative over the empire tier above):
@migr_resettle_factor_nomadic	= 0.5	# trait_nomadic — used to moving (vanilla cost mult -0.35)
@migr_resettle_factor_sedentary	= 1.5	# trait_sedentary — hates being uprooted (vanilla +0.35)

# --- Empire-size scaling: keeps the surcharge relevant late-game instead of going flat as
#     income inflates. Same SHAPE as vanilla tech/tradition cost growth:
#       multiplier = 1 + max(0, empire_size - SIZE_BASE) * SIZE_PENALTY
#     SIZE_BASE matches vanilla's EMPIRE_SIZE_BASE (100, the no-penalty threshold). SIZE_PENALTY is
#     OUR OWN variable (independent of vanilla's EMPIRE_SIZE_TECH_COST_PENALTY = 0.002), deliberately
#     STEEPENED to 0.01 (5x vanilla) because our surcharge has a flat base (unlike tech, whose base
#     cost itself balloons through the tree). At 0.01: size 300 -> x3, size 600 -> x6 — so bulk
#     resettlement in a sprawling late-game empire actually bites. Tunable; re-check after playtest.
@migr_resettle_size_base		= 100
@migr_resettle_size_penalty		= 0.01

# --- Settling-in time penalty (the "takes time" half). A timed productivity/happiness
#     debuff on the resettled pops, simulating travel + adjustment since the engine has no
#     native travel-time mechanic. Waived for gestalt empires and nomadic species. ---
@migr_resettle_penalty_days		= 1800	# ~5 years of "Recent Relocation"
@migr_resettle_happiness		= -0.15
@migr_resettle_workforce		= -0.15	# reduced bonus workforce = lower output while settling
