# Economy Overhaul — slice 3: multiplier taming (civic overrides)
# ===============================================================
# Two pickable civics become near-auto-picks once space is the primary economy
# (design doc "tame the multipliers", levers #10-#11). Both are handled by
# redefining the vanilla civic KEY (civics are keyed DB objects; our mod loads
# after vanilla so the redefinition wins — the zzz_ prefix also makes that explicit).
# Logged in docs/compatibility.md.
#
# Each is a FAITHFUL copy of vanilla 4.4.3 with only the marked change. We are
# version-pinned to 4.4.3, so copy-staleness is a non-issue; re-sync if the pin moves.
# ⚠️ RUNTIME-VERIFY: error.log clean on load; Astro-Mining absent from empire
# creation; Privatized Exploration shows +10% (not +25%) station output.

# --- #10 Astro-Mining Drones: DISABLED from selection ------------------------
# Vanilla grants station_gatherers_produces_mult = 0.5 (a double-compounder with our
# station buffs) plus a planet self-nerf. Cutting the buff would make it a trap pick
# (planet nerf with no upside), so per the design decision (user-approved) we DISABLE
# it instead — playable/ai_playable -> { always = no }. Everything else is left intact
# so the civic is correct if ever re-enabled. (vanilla: 02_gestalt_civics.txt)
civic_machine_astromining_drones = {
	playable = {
		always = no		# ECON OVERHAUL: disabled (was: has_megacorp = yes)
	}
	ai_playable = {
		always = no		# ECON OVERHAUL: disabled (was: has_megacorp = yes)
	}
	potential = {
		authority = { value = auth_machine_intelligence }
	}

	possible = {
		origin = {
			NOT = {
				value = origin_machine
			}
		}
	}

	random_weight = {
		base = @civic_rare_random_weight
	}

	ai_weight = {
		base = @ai_civic_uncommon_base_weight
	}
	swap_type = {
		name = civic_machine_astromining_drones
		description = "civic_tooltip_machine_astromining_drones_effects"
		trigger = { is_nomadic = no }
		modifier = {
			planet_miners_minerals_produces_add = -2
			planet_technician_energy_produces_add = -1
			station_gatherers_produces_mult = 0.5
			country_starbase_capacity_mult = 0.50
			stations_cost_mult = -0.50
			planet_jobs_simple_drone_produces_mult = -0.50
			councilor_gestalt_growth_exp_gain = @gestalt_civic_node_xp_rate
			add_attunement_the_composer_of_strands = @civic_attunement1
		}
	}
	swap_type = {
		name = civic_machine_astromining_drones
		trigger = { is_nomadic = yes }
		modifier = {
			planet_technician_energy_produces_add = -2
			starbase_collected_system_resources_mult = 0.5
			country_starbase_capacity_mult = 0.50
			megastructure_waystation_megastructure_build_speed_mult = 0.50
			planet_jobs_simple_drone_produces_mult = -0.50
			councilor_gestalt_growth_exp_gain = @gestalt_civic_node_xp_rate
			add_attunement_the_composer_of_strands = @civic_attunement1
		}
	}
}

# --- #11 Privatized Exploration: station bonuses CUT +0.25 -> +0.10 -----------
# Pure-bonus corporate civic, so cutting (not disabling) keeps it a balanced pick.
# Only the two station_*_produces_mult lines change; starbase build-speed bonuses
# and all gating are left as vanilla. (vanilla: 03_corporate_civics.txt)
civic_privatized_exploration = {
	playable = { has_first_contact_dlc = yes }

	description = "civic_privatized_exploration_effects"
	negative_description = "civic_eager_explorers_negative_effects"

	potential = {
		OR = {
			authority = { value = auth_corporate }
			civics = { value = civic_galactic_sovereign_megacorp }
		}
		ethics = {
			NOT = {
				value = ethic_gestalt_consciousness
			}
		}
	}
	modification = {
		add = {
			has_technology = tech_subspace_drive
		}
		remove = {
			has_technology = tech_jump_drive_1
		}
	}
	alternate_civic_version = civic_eager_explorers
	possible = {
		is_nomadic = no
		graphical_culture = {
			NOR = {
				text = graphical_culture_tooltip_not_biological
				value = biogenesis_01
				value = biogenesis_02
			}
		}
		origin = {
			NOR = {
				# No FC origins.
				value = origin_payback
				value = origin_broken_shackles
				value = origin_fear_of_the_dark
				value = origin_synthetic_fertility
				value = origin_unplugged
			}
		}
		civics = {
			NOT = {
				value = civic_eager_explorers
			}
			NOR = {
				text = civic_tooltip_not_guardian
				value = civic_sovereign_guardianship
				value = civic_corporate_sovereign_guardianship
				value = civic_hive_sovereign_guardianship
				value = civic_machine_sovereign_guardianship
			}
			NOR = {
				text = civic_tooltip_not_hyperspace_specialty
				value = civic_hyperspace_specialty
				value = civic_corporate_hyperspace_specialty
				value = civic_hive_hyperspace_specialty
				value = civic_machine_hyperspace_specialty
			}
			NOR = {
				text = civic_tooltip_not_guided
				value = civic_guided_sapience
				value = civic_corporate_guided_sapience
				value = civic_hive_guided_sapience
				value = civic_machine_guided_sapience
			}
		}
	}

	modifier = {
		starbase_building_build_speed_mult = 0.3
		starbase_module_build_speed_mult = 0.3
		starbase_upgrade_speed_mult = 0.3
		station_gatherers_produces_mult = 0.10		# ECON OVERHAUL: was 0.25
		station_researchers_produces_mult = 0.10	# ECON OVERHAUL: was 0.25
	}

	random_weight = {
		base = 0
	}
}
