# Economy Overhaul — "Space > Planets" country static modifier
# ------------------------------------------------------------
# A single permanent COUNTRY modifier applied to every empire at game start
# (see econ_on_actions.txt). Zero vanilla file overrides — this is a NEW static
# modifier, applied additively, exactly like migration_overhaul's approach.
#
# Why these modifier names work at country scope:
#   - station_gatherers_produces_mult / station_researchers_produces_mult are
#     country-scope economy-category modifiers (vanilla uses them in difficulty
#     static modifiers, e.g. difficulty_grand_admiral).
#   - planet_<job>_<resource>_produces_mult are the SAME modifiers vanilla
#     economy techs apply at country scope (e.g. planet_miners_minerals_produces_mult
#     in 00_eng_tech.txt). A country-scope modifier cascades to all owned planets.
#
# Numbers come from econ_space_economy_variables.txt. hide_from_country_list = no
# so players can see the effect in their country modifier tooltip.

econ_space_primacy = {
	hide_from_country_list = no

	# v2 (2026-06-23): the flat +50% station_*_produces_mult lines were REMOVED.
	# "Space is the primary economy" is now carried by BUFFED DEPOSIT YIELDS at the
	# source (whole-file 01_orbital_deposits.txt, per-resource produces mults: minerals ×1.40 /
	# energy ×1.60 / research ×1.15, v2.5) + the station
	# repeatables (slice 2), so we don't double-count base × flat-mult × repeatable
	# (the multiplier-compounding the design doc warned about). This modifier now
	# carries only the planet-DOWN structural support that has no cleaner home.

	# Housing scarcity (v2.1): the global planet_housing_mult was REPLACED by a targeted
	# -30% cut to URBAN (city/hive/nexus) district housing (whole-file 00_urban_districts.txt).
	# Reason: the global mult also cut RURAL housing → penalised wide/resource planets; the
	# urban-only cut spares rural and squeezes TALL/specialist-heavy planets (which stack city
	# districts) → "wide > tall". Overcrowding defines (zzz_econ_defines.txt) still make it bite.

	# NOTE (v2.3): the pop ASSEMBLY nerf moved OUT of this universal modifier into a
	# separate, machine-EXEMPT modifier (econ_organic_assembly_nerf below), applied
	# conditionally in event econ_overhaul.1. Rationale: machine empires grow ONLY via
	# assembly and are already heavily reined late-game by the vanilla Country Growth
	# Scale (empire-pop-count tax on the flat channel) — stacking our -33% on top guts
	# them late. Organic empires, by contrast, can pair robot ASSEMBLY with their organic
	# LOGISTIC growth (dual-channel), so they DO need the assembly tax to prevent abuse.
	# Hence: no assembly nerf for machines, keep it for organics (hives have no assembly,
	# so the gate harmlessly includes them). See econ_organic_assembly_nerf.

	# Pop GROWTH nerf (v2.2): the flat-additive growth channel the assembly nerf above
	# MISSED. Hive spawning drones / plantoid-lithoid budding / clone vats emit
	# `bonus_pop_growth` (flat, doesn't slow as a planet fills) rather than
	# planet_pop_assembly_*. bonus_pop_growth_mult scales ONLY that flat channel and
	# does NOT touch logistic_growth_mult, so base organic growth + the Fertile / Rapid
	# Breeders traits stay vanilla. Cascades from this country modifier to every planet
	# (vanilla cybernetic/bio-hive authorities use bonus_pop_growth_mult at country scope).
	bonus_pop_growth_mult = @econ_growth_nerf

	# Slice 3 (lever #12b): trim the scaling-kilostructure build caps. *_limit_add is
	# a country modifier (category = country); the cap = base 0 + sum of these, so a
	# negative value lowers it for every empire symmetrically. Per-tier kilostructure
	# OUTPUT is cut separately via the 07_scripted_variables_machine_age.txt replacement.
	arc_furnace_limit_add = @econ_kilostructure_limit_reduction
	dyson_swarm_limit_add = @econ_kilostructure_limit_reduction
}

# Pop ASSEMBLY nerf — applied ONLY to NON-machine empires (see econ_overhaul.1 gate:
# is_machine_empire = no). Machines grow solely by assembly and are already late-game
# capped by vanilla Country Growth Scale, so they are exempt. Organic empires keep this
# so robot assembly can't be stacked tax-free on top of their organic logistic growth.
# Hives are included by the gate but have no assembly, so it's a harmless no-op for them.
econ_organic_assembly_nerf = {
	hide_from_country_list = no
	planet_pop_assembly_mult = @econ_assembly_nerf
}

# AI-only planetary relief — applied ONLY to AI empires (see econ_overhaul.1 gate:
# is_ai = yes). The AI cannot perceive our overcrowding growth-stall (its planner is
# deficit-driven: it reacts to homeless pops, not to growth silently halting BEFORE
# homelessness), so it never builds residences and stalls at planet capacity. This
# restores housing so AI empires stay economically functional. All 7 players are human,
# so the human-facing housing scarcity (urban -30% + tightened overcrowding) is intact.
econ_ai_planet_relief = {
	hide_from_country_list = no
	planet_housing_mult = @econ_ai_housing_relief
}
