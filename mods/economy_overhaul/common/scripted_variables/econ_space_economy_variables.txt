# Economy Overhaul — "Space > Planets" tunable values
# ---------------------------------------------------
# All balance knobs for the space-vs-planet rebalance live here so the batched
# in-game test session can tune them in ONE place. These feed the country static
# modifier econ_space_primacy (see econ_space_economy_modifiers.txt), applied to
# every empire at game start (econ_on_actions.txt). Symmetric across all empires
# => MP-fair (no single player gains an advantage).
#
# Sign convention: buffs are positive, nerfs are negative. Values are mults
# (0.5 = +50%, -0.5 = -50%).

# --- Space strength: now carried by BUFFED DEPOSIT YIELDS, not a flat modifier ---
# v2 (2026-06-23): the old flat +50% station_*_produces_mult modifier was REMOVED to
# avoid base × flat-mult × repeatable compounding. "Space is primary" is now intrinsic
# to the deposits: common/deposits/01_orbital_deposits.txt is whole-file-replaced with
# every `produces` value ×1.75 (tune that multiplier there). Station repeatables
# (slice 2, @econ_repeatable_per_level below) still scale space output over time.

# --- Housing scarcity (v2.1) — now done in the district files, not here ---
# Urban (city/hive/nexus) district housing is cut -30% directly in the whole-file
# replacement common/districts/00_urban_districts.txt (tune the 0.70 multiplier there).
# This replaced the old global @econ_housing_nerf / planet_housing_mult, which also hit
# rural housing and penalised wide play. Rural housing stays vanilla. Pairs with the
# tightened overcrowding defines (zzz_econ_defines.txt). #1 calibration target.

# --- Pop ASSEMBLY nerf (v2: bring assembly in line with logistic organic growth) ---
# Assembly is a FLAT rate (doesn't slow as a planet fills), organic growth is LOGISTIC
# (slows near capacity). v2.3: this nerf is now applied ONLY to NON-machine empires
# (via econ_organic_assembly_nerf, gated is_machine_empire = no in econ_overhaul.1).
# Why: machine empires grow SOLELY by assembly and are already heavily late-capped by the
# vanilla Country Growth Scale (an empire-pop-count tax on the flat channel that EXEMPTS
# logistic — confirmed in-game) — stacking our -33% on top guts them late, which we don't
# want. Organic empires keep it so robot ASSEMBLY can't be stacked tax-free on top of
# their organic LOGISTIC growth (the dual-channel abuse). Hives are inside the gate but
# have no assembly, so it's a no-op for them.
@econ_assembly_nerf = -0.33				# -33% pop assembly, NON-machine empires only (organics; machines exempt)

# --- Pop GROWTH nerf (v2.2: the spawning-pool / flat-growth parity gap) ---
# The assembly nerf above only touches planet_pop_assembly_* (machine/robot ASSEMBLY).
# Hive spawning drones, plantoid/lithoid BUDDING traits, and clone vats instead emit
# `bonus_pop_growth` — a FLAT additive growth that, like assembly, does NOT slow as a
# planet fills (unlike base logistic growth). Verified in-game (2026-06-24) that our
# assembly nerf missed this channel, so hives kept out-growing nerfed assembly empires.
# bonus_pop_growth_mult scales ONLY that flat-additive channel. It does NOT touch
# logistic_growth_mult, so base organic growth AND the Fertile / Rapid Breeders traits
# (both logistic_growth_mult) stay at vanilla as the reference curve. (The design doc's
# earlier worry that this hits Fertile was based on a wrong assumption — verified false.)
# Set equal to the assembly nerf so every FLAT-growth channel lands on one line; tune
# independently here if hive/budding/clone economies need a different slope than assembly.
@econ_growth_nerf = -0.33				# -33% flat additive pop growth (spawning pools, budding, clone vats), all empires

# --- Planetary primary-resource job output NERFS (OPTIONAL fine-tune — DEFAULT 0) ---
# SUPERSEDED as the primary planet-down lever. The structural slice
# (whole-file replacements: 00_scripted_variables.txt planet size cap 25->18 +
# 100_scripted_variables_zones.txt rural jobs 200->160) now does the planet-down work. Per the design doc (Track 1), this flat per-pop
# nerf becomes an OPTIONAL light fine-tune that must NOT stack hard on the
# structural cuts (double-nerf -> unemployment death spiral). Left at 0 (unused) —
# the modifier lines are removed from econ_space_primacy; re-wire them there only
# if playtest shows planets still over-produce AFTER the structural cuts.
# Per-resource variants are exactly what vanilla economy techs use at country scope
# (00_eng_tech.txt / 00_phys_tech.txt / 00_soc_tech.txt), so they cascade to all
# owned planets. Research deliberately never nerfed (vision targets PRIMARY resources).
@econ_planet_miners_nerf = 0			# fine-tune lever (unused); e.g. -0.2 = -20% planetary minerals
@econ_planet_technician_nerf = 0		# fine-tune lever (unused); e.g. -0.2 = -20% planetary energy
@econ_planet_farmers_nerf = 0			# fine-tune lever (unused); e.g. -0.2 = -20% planetary food

# --- Slice 2: bulk scaling PARITY (repeatable-tech per-level rate) ---
# ONE rate shared by all five repeatables (3 vanilla tile overrides + 2 new station
# techs) so planet-job scaling and station scaling "climb together but gently"
# (design doc Track 1, levers 2-3). Vanilla tile repeatables are 0.05/level and
# INFINITE (levels = -1) with NO station counterpart — that asymmetry is why space
# falls behind. We (a) NERF the tile rate 0.05 -> 0.03 (bloat control, Cetus-safe)
# and (b) ADD station gatherers/researchers repeatables at the SAME 0.03 so the
# space>planet RATIO set by slice 1 (structural cuts + station buffs) is preserved
# as the game advances rather than eroding. Lower this further if late-game numbers
# still bloat; raise if scaling feels too flat. See tech_repeatable_* in
# common/technology/zzz_econ_repeatable_techs.txt.
@econ_repeatable_per_level = 0.03		# +3% per repeatable level (vanilla tile was +5%)

# --- Slice 3: kilostructure build-limit reduction (multiplier taming, lever #12b) ---
# The Arc Furnace / Dyson Swarm build caps are `base 0 + sum of modifier:*_limit_add`
# (verified: common/script_values/04_script_values_machine_age.txt). So we lower the cap
# cleanly with a NEGATIVE country modifier (additive, no vanilla override) applied to
# every empire — fewer of these stacking compounders. Vanilla grants total ~5-6 per
# structure (megastructure tech + machine-age tech + an origin); -1 trims one slot.
# Empires without the unlocking tech have cap 0 and still can't build (harmless).
@econ_kilostructure_limit_reduction = -1	# -1 to each of arc-furnace / dyson-swarm caps
